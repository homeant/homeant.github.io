	ACT_RE_PROCDEF.KEY_ = ?;

SELECT
	art.ID_ TASK_ID_,
	art.EXECUTION_ID_,
	art.PROC_INST_ID_,
	art.NAME_,
	art.ASSIGNEE_,
	art.CREATE_TIME_,
	are.PARENT_ID_,
	are.PROC_DEF_ID_,
	are.ROOT_PROC_INST_ID_,
	are.ACT_ID_,
	are.START_TIME_ 
FROM
	ACT_RU_TASK art
	LEFT JOIN ACT_RU_EXECUTION are ON art.EXECUTION_ID_ = are.ID_ 
WHERE
	art.PROC_DEF_ID_ = ?;

SELECT
	arv.ID_,
	arv.TYPE_,
	arv.NAME_,
CASE
		arv.TYPE_ 
		WHEN 'string' THEN
		arv.TEXT_ 
		WHEN 'boolean' THEN
		arv.LONG_ 
		WHEN 'integer' THEN
		arv.LONG_ ELSE '' 
	END VALUE_
	FROM
		ACT_RU_VARIABLE arv 
WHERE
	arv.EXECUTION_ID_ = ?;
